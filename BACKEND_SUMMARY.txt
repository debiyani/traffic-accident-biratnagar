// 1. Get dropdown options
async function loadDropdowns() {
  const res = await fetch('http://localhost:5000/api/options');
  const data = await res.json();
  
  // Populate ward dropdown
  document.getElementById('wardSelect').innerHTML = 
    data.wards.map(w => `<option>${w}</option>`).join('');
  
  // Populate location dropdown
  document.getElementById('locationSelect').innerHTML = 
    data.locations.map(l => `<option>${l}</option>`).join('');
  
  // ... repeat for time_ranges, road_types, months
}

// 2. Analyze accident rate
async function analyzeAccidents() {
  const response = await fetch('http://localhost:5000/api/analyze', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      time_range: document.getElementById('timeRange').value,
      ward: parseInt(document.getElementById('ward').value),
      location: document.getElementById('location').value,
      month: parseInt(document.getElementById('month').value),
      road_type: document.getElementById('roadType').value
    })
  });
  
  const result = await response.json();
  
  // Display accident rate
  document.getElementById('accidentRate').textContent = 
    result.with_road_type.total_accidents;
  
  // Display severity
  const severity = result.with_road_type.severity_distribution;
  document.getElementById('highSeverityPct').textContent = 
    severity.high_pct + '%';
  
  // Draw pie chart with seasonal data
  drawPieChart(
    result.seasonal_breakdown.seasonal_breakdown.map(s => s.season),
    result.seasonal_breakdown.seasonal_breakdown.map(s => s.accidents)
  );
}

// 3. Predict severity
async function predictSeverity() {
  const response = await fetch('http://localhost:5000/api/predict-severity', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      time_range: document.getElementById('timeRange').value,
      ward: parseInt(document.getElementById('ward').value),
      location: document.getElementById('location').value,
      month: parseInt(document.getElementById('month').value),
      road_type: document.getElementById('roadType').value
    })
  });
  
  const result = await response.json();
  
  // Display prediction
  document.getElementById('severity').textContent = result.prediction.toUpperCase();
  document.getElementById('riskLevel').textContent = result.risk_level;
  document.getElementById('probability').textContent = 
    (result.probability_high * 100).toFixed(1) + '%';
  document.getElementById('recommendation').textContent = result.recommendation;
  
  // Color code by risk
  const element = document.getElementById('riskLevel');
  element.style.color = result.risk_level === 'HIGH' ? 'red' : 'green';
}
