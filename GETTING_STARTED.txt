📱 FRONTEND INTEGRATION (Copy-Paste Ready)
═══════════════════════════════════════════════════════════════════════════════

Add to your HTML:

<script>
const API = 'http://localhost:5000/api';

// 1. Load dropdown options on page load
async function loadOptions() {
  const res = await fetch(`${API}/options`);
  const data = await res.json();
  
  document.getElementById('ward').innerHTML = 
    data.wards.map(w => `<option>${w}</option>`).join('');
  document.getElementById('location').innerHTML = 
    data.locations.map(l => `<option>${l}</option>`).join('');
  document.getElementById('timeRange').innerHTML = 
    data.time_ranges.map(t => `<option>${t}</option>`).join('');
  document.getElementById('roadType').innerHTML = 
    data.road_types.map(r => `<option>${r}</option>`).join('');
  document.getElementById('month').innerHTML = 
    data.months.map(m => `<option value="${m.num}">${m.name}</option>`).join('');
}

// 2. When user submits form
async function analyzeAndPredict() {
  const body = {
    time_range: document.getElementById('timeRange').value,
    ward: parseInt(document.getElementById('ward').value),
    location: document.getElementById('location').value,
    month: parseInt(document.getElementById('month').value),
    road_type: document.getElementById('roadType').value
  };
  
  // Get accident analysis
  const analysisRes = await fetch(`${API}/analyze`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body)
  });
  const analysis = await analysisRes.json();
  
  // Display accident rate
  document.getElementById('accidentRate').textContent = 
    analysis.with_road_type.total_accidents + ' accidents';
  
  // Display severity
  const severity = analysis.with_road_type.severity_distribution;
  document.getElementById('severity').innerHTML = `
    Low: ${severity.low_pct}% | 
    Medium: ${severity.medium_pct}% | 
    High: ${severity.high_pct}%
  `;
  
  // Draw pie chart with seasonal data
  const seasonal = analysis.seasonal_breakdown.seasonal_breakdown;
  drawPieChart(
    seasonal.map(s => s.season),
    seasonal.map(s => s.accidents)
  );
  
  // Get ML prediction
  const predRes = await fetch(`${API}/predict-severity`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body)
  });
  const prediction = await predRes.json();
  
  // Display prediction
  const riskColor = prediction.risk_level === 'HIGH' ? 'red' : 'green';
  document.getElementById('riskLevel').innerHTML = 
    `<span style="color: ${riskColor};">${prediction.risk_level}</span>`;
  document.getElementById('recommendation').textContent = 
    prediction.recommendation;
}

// Call on page load
loadOptions();
</script>
